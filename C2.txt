#include <stdio.h>

void foo(int *pokazivac) {
    printf("Vrijednost nove varijable pokazivac: %p\n", pokazivac);
    printf("Adresa nove varijable pokazivac: %p\n", &pokazivac);
    printf("Vrijednost na adresi na koju pokazuje varijabla pokazivac: %d\n", *pokazivac);

}


int main()
{
    int a = 10;
    int *pa;
    pa = &a;
    
    printf("Vrijednost od a je %d, u adresi %p \n", a, &a);
    printf("Vrijednost od pa je %p, u adresi %p \n", pa, &pa);
    printf("Kod: '*pa' ce ispisati vrijednost a: %d\n", *pa);
    
    printf("\n-----\n\n");
    
    foo(pa);

}
